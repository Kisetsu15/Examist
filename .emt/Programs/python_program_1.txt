def transfer(balance, amount):
    if amount <= 0:
        print("Invalid amount")
        return

    if amount > balance:
        print("Insufficient balance")
        return

    if amount % 100 != 0:
        print("Amount should be multiple of 100")
        return

    balance -= amount

    print("Transferred:", amount)
    print("Remaining balance:", balance)
transfer(1000, 800)
